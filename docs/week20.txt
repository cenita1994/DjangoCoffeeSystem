from django import forms
from .models import Product

class ProductForm(forms.ModelForm):
    class Meta:
        model = Product
        fields = ['name', 'category', 'price', 'quantity', 'description']

Step 4: Open inventory/views.py
from django.shortcuts import render, redirect
from .models import Product
from .forms import ProductForm

def home(request):
    return render(request, 'home.html')

def product(request):
    products = Product.objects.all()
    return render(request, 'inventory/product_list.html', {'products': products})

def add_product(request):
    form = ProductForm()

    if request.method == 'POST':
        form = ProductForm(request.POST)
        if form.is_valid():
            form.save()
            return redirect('product')

    return render(request, 'inventory/product_form.html', {'form': form})

def about(request):
    return render(request, 'inventory/about.html')

Step 7: inventory/views.py (add)

from django.shortcuts import render, redirect, get_object_or_404


def edit_product(request, id):
    product_item = get_object_or_404(Product, id=id)
    form = ProductForm(instance=product_item)

    if request.method == 'POST':
        form = ProductForm(request.POST, instance=product_item)
        if form.is_valid():
            form.save()
            return redirect('product')

    return render(request, 'inventory/product_form.html', {'form': form})

Step 10: inventory/views.py (add)

def delete_product(request, id):
    product_item = get_object_or_404(Product, id=id)

    if request.method == 'POST':
        product_item.delete()
        return redirect('product')

    return render(request, 'inventory/delete.html', {'product': product_item})
